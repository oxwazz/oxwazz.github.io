    // do something

## Loop
```kotlin
for (i in 0..10) {
    // do something
}

for (i in 0 until 10) {
    // do something
}

for (i in 10 downTo 0) {
    // do something
}

for (i in 0..10 step 2) {
    // do something
}
```

## Input
```kotlin
// Button
Button(onClick = { /* Do something */ }) {
    Text("Click me")
}

// TextField
var text by remember { mutableStateOf("") }
    TextField(
        value = text,
        onValueChange = { text = it },
        singleLine = true,
        label = { Text(stringResource(R.string.bill_amount)) },
        keyboardOptions = KeyboardOptions(keyboardType = KeyboardType.Number),
        modifier = modifier
    )
```

## keyboardOptions
```kotlin
// KeyboardOptions
TextField(
    value = text,
    onValueChange = { text = it },
    keyboardOptions = KeyboardOptions.Default.copy(  // use other default options
        keyboardType = KeyboardType.Number,          // Number, Text, Phone, Email, Password, etc
        imeAction = ImeAction.Done                   // Done, Next, Search, Go, Send, etc
    )
)
```

## Annotation
```kotlin
// @Composable
@Composable
fun MyComposable() {
    Text("Hello, World!")
}

// @Preview
@Preview
@Composable
fun MyComposablePreview() {
    MyComposable()
}

// @StringRes
@StringRes
fun getStringRes(@StringRes resId: Int): String {
    return stringResource(resId)
}
```
